TableElement.get = function(name) {
  var names = name.split('.');
  var table = names[0];
  var tableDef = Table.get(table);
  var e = null;
  for (var i = 1; i < names.length; i++) {
    e = tableDef.getElement(names[i]);
    if (i == names.length - 1)
      break;
    if (!e.isReference())
      break;
    tableDef = Table.get(e.getReference());
  }
  return e;
};
